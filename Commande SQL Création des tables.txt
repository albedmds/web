CREATE TABLE Entreprise
(
    Adresse_Entreprise  LONGCHAR,
    Nom_Entreprise  LONGCHAR,
    Code_Entreprise INTEGER,
    PRIMARY KEY (Code_Entreprise)
);

CREATE TABLE Info_supplementaire_Perso
(
    Id_Perso    INTEGER DEFAULT VALUE 0,
    Id_Entreprise   INTEGER DEFAULT VALUE 0,
    Code_Info_Sup_Perso INTEGER,
    PRIMARY KEY (Code_Info_Sup_Perso)
);

CREATE TABLE Media
(
    Donnee_Media    LONGCHAR,
    ID_Perso_Publieur   INTEGER DEFAULT VALUE 0,
    Code_Media  INTEGER,
    Type_Media  VARCHAR(255),
    Horaire_Publication DATE,
    PRIMARY KEY (Code_Media)
);

CREATE TABLE Message
(
    Code_Message    INTEGER,
    Horaire_Envoi   DATE,
    Id_Perso_1M INTEGER DEFAULT VALUE 0,
    Id_Perso_2M INTEGER DEFAULT VALUE 0,
    Accuse_Reception    BIT DEFAULT VALUE No,
    Accuse_Lecture  BIT DEFAULT VALUE No,
    Donnee_Message  LONGCHAR,
    PRIMARY KEY (Code_Message)
);

CREATE TABLE Offre_Emploi
(
    Intitulé    VARCHAR(255),
    Id_Entreprise   INTEGER DEFAULT VALUE 0,
    Code_Offre_Emploi   INTEGER,
    PRIMARY KEY (Code_Offre_Emploi)
);

CREATE TABLE Parametre
(
    Numero_Media    INTEGER DEFAULT VALUE 0,
    Code_Parametre  INTEGER,
    Degre_Confidentialite   INTEGER DEFAULT VALUE 0,
    PRIMARY KEY (Code_Parametre)
);

CREATE TABLE Personne
(
    Code_Personne   INTEGER,
    Nom LONGCHAR,
    Prenom  LONGCHAR,
    Pseudo  LONGCHAR,
    Email   LONGCHAR,
    Mot_de_passe    LONGCHAR,
    PRIMARY KEY (Code_Personne)
);

CREATE TABLE Relation
(
    Id_Perso_1  INTEGER DEFAULT VALUE 0,
    Id_Perso_2  INTEGER DEFAULT VALUE 0,
    Horaire_Relation    DATE,
    Code_Relation   INTEGER,
    PRIMARY KEY (Code_Relation)
);

CREATE TABLE Type_Relation
(
    Id_Utilisateur  INTEGER DEFAULT VALUE 0,
    Id_Relation INTEGER DEFAULT VALUE 0,
    Degre_Relation  INTEGER DEFAULT VALUE 0,
    Code_Type_Relation  INTEGER,
    PRIMARY KEY (Code_Type_Relation)
);

CREATE TABLE Utilisateur
(
    Numero_Perso    INTEGER DEFAULT VALUE 0,
    Droit   VARCHAR(255),
    Code_Utilisateur    INTEGER,
    PRIMARY KEY (Code_Utilisateur)
);
